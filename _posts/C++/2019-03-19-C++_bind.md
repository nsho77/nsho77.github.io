# boost::bind 가 하는 일은?
 - 임의의 함수, 함수 포인터, 함수 객체, 멤버 함수를 함수 객체로 만들 수 있다.
 - 원하는 위치에 원하는 값을 전달 시킬 수 있는 함수 객체를 만든다.

{% highlight cpp %}
#include <boost/bind.hpp>
class A
{
public:
    void Start();
};
std::vector<A> v;
///... (add to v) ...
std::for_each(v.begin(), v.end(), std::bind(&A::Start,_1));

{% endhighlight %}
